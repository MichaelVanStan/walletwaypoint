const states = defineCollection({

    brackets: s.object({ single, married, head }).optional(),
    flatRate: s.number().optional(),
    standardDeductions: s.object({ single, married, head }).optional(),
    personalExemptions: s.object({ single, married, dependent }).optional(),
    topRate: s.number(),
    numberOfBrackets: s.number(),
    editorialTitle: s.string(),

    editorialContent: s.string(),

    tips: s.array(s.string()),
