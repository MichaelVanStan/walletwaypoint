```typescript
'use client';

import { Select, SelectTrigger, SelectValue, SelectContent, SelectItem } from '@/components/ui/select';
import { Button } from '@/components/ui/button';
import type { FilterConfig } from '@/lib/compare/filter-config';
```

Props:
```typescript
interface ComparisonFiltersProps {
  filters: FilterConfig[];
  activeFilters: Record<string, string>;
  onFilterChange: (filterId: string, value: string | null) => void;
  onClearAll: () => void;
}
```

- Wraps in `<nav aria-label="Product filters">` (per UI-SPEC Accessibility Contract)
- Renders horizontal row on desktop (`flex flex-wrap items-center gap-4`), stacked on mobile (`flex flex-col gap-4 lg:flex-row lg:flex-wrap lg:items-center`)
- Label: `<span className="text-sm font-medium text-foreground">Filter by:</span>`
- Each filter renders a `<Select>` with `<SelectTrigger size="sm">` containing `<SelectValue placeholder={filter.label} />`
- CRITICAL Base UI Select API: `value` prop is the active filter value. `onValueChange` receives `(value: string | null, event)` -- handle null by resetting to 'all'. The filter value from `activeFilters[filter.id]` maps to Select value.
- "Clear filters" button: Only visible when at least one filter is not 'all'. Renders `<Button variant="ghost" size="sm" onClick={onClearAll}>Clear filters</Button>` with `aria-label="Clear all filters"`.
